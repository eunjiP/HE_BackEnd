CREATE TABLE `product` (
	`i_product`	INT UNSIGNED AUTO_INCREMENT,
	`productName` VARCHAR(50)	NOT NULL,
	`productImg` VARCHAR(500) NOT NULL,
	`ctnt` VARCHAR(500) NOT NULL,
	`cate` INT UNSIGNED NOT NULL,
	PRIMARY KEY(i_product)
);

CREATE TABLE `region` (
	`rootCode`	INT UNSIGNED	NOT NULL PRIMARY KEY,
	`regionNm`	VARCHAR(10)	NOT NULL
);

CREATE TABLE `subregion` (
	`subCode`	INT UNSIGNED	NOT NULL PRIMARY KEY,
	`rootCode`	INT UNSIGNED	NOT NULL,
	`subNm`	VARCHAR(50)	NOT NULL,
	`regionUrl`	VARCHAR(500)	NOT NULL
);

CREATE TABLE `user` (
	`i_user`	INT UNSIGNED NOT NULL	AUTO_INCREMENT PRIMARY KEY,
	`uuid`	INT UNSIGNED NOT NULL	UNIQUE,
	`regionCode`	VARCHAR(10)	NULL
);

CREATE TABLE `cate` (
	`i_product`	INT UNSIGNED NOT NULL PRIMARY KEY,
	`cate1Nm`	VARCHAR(20)	NOT NULL
);

CREATE TABLE `qna` (
	`i_board`	INT UNSIGNED NOT NULL AUTO_INCREMENT PRIMARY KEY,
	`i_user`	INT UNSIGNED NOT NULL,
	`title`	VARCHAR(500)	NOT NULL,
	`ctnt`	VARCHAR(1000)	NULL,
	`created_at`	DATETIME	NULL	DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE `comment` (
	`i_cmt`	INT UNSIGNED NOT NULL PRIMARY KEY AUTO_INCREMENT,
	`cmt`	VARCHAR(500)	NOT NULL,
	`i_board`	INT UNSIGNED	NOT NULL,
	`i_user`	INT UNSIGNED NOT NULL
);
